term

   Suporte de Terminal

[term.clear] term.clrscr(): apaga a tela

 
[term.cleareol] term.clreol(): apaga a partir da posição corrente do cursor até o fim da linha

 

[term.moveto] term.gotoxy( x, y ): posiciona o cursor em uma coordenada específica


[term.moveup] term.up( delta ): move o cursor para cima "delta" linhas

 

[term.movedown] term.down( delta ): move o cursor para baixo "delta" linhas

 

[term.moveleft] term.left( delta ): move o cursor para esquerda "delta" caracteres

 
[term.moveright] term.right( delta ): move o cursor para direita "delta" caracteres

 

[term.getlinecount] Lines = term.lines(): retorna o número de linhas

 

[term.getcolcount] Cols = term.cols(): retorna o número de colunas

 


[term.printstr] term.putstr( s1, s2, ... ): imprime um string especificado no terminal

[term.printchar] term.put( c1, c2, ... ): writes the imprime um carater específico no terminal

 

[term.getx] Cx = term.cursorx(): retorna a posição X do cursor

 

[term.gety] Cy = term.cursory(): retorna a posição Y do cursor

 

[term.inputchar] c = term.getch( term.WAIT | term.NOWAIT ): retorna um caracter lido do terminal
